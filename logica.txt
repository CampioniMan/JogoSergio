// faz o flappy pular
// apenas faz com que o índice x do irDescendo seja alterado
function pular()
{
	indice = 0;
}

// faz o Flappy descer cada vez mais
// baseia-se numa função e num índice(x) da função
function irDescendo()
{
	var x = funcaum(); // usa o índice
	flappy.y += x;
	indice++;
}

// a função que devolve um número representativo do índice
function funcaum()
{
	var aux = -((indice^2)/25) + indice;
	flappy.y += aux
}

// quando o cano chega no final, ele volta pro começo

function canoAndar()
